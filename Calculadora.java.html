public class Calculadora {
    //caracteristicas
    public double num1;
    public double num2;
    public double resultado;

    //construtor
    public Calculadora(){}

    //Ações
    public double somar(double n1, double n2){
        num1 = n1;
        num2 = n2;
        resultado = n1 + n2;
        return resultado;
    }
}
